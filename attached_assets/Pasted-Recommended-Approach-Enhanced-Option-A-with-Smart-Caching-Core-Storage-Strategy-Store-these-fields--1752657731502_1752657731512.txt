CREATE TABLE jira_tickets (
    id SERIAL PRIMARY KEY,

    summary TEXT,

    jira_created_at TIMESTAMP,
    jira_updated_at TIMESTAMP,
    last_synced_at TIMESTAMP,
    content_hash VARCHAR(64), -- SHA-256 of concatenated fields
